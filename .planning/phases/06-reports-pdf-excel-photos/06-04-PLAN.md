Implementação do submit:
```typescript
async function handleSubmit(e: FormEvent) {
  e.preventDefault();
  setLoading(true);
  try {
    const res = await fetch('/api/relatorios', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ formato, filtros: { dataInicio, dataFim, obraId, cameraId, classificacao, placa } }),
    });
    const data = await res.json();
    if (!res.ok) { setError(data.error); return; }
    onReportRequested(data.relatorioId);
    // resetar form
  } finally {
    setLoading(false);
  }
}
```

**Socket.IO integration** (REPORTS-06):
```typescript
useEffect(() => {
  // Reutilizar o socket existente do projeto (verificar como outros componentes conectam)
  // Padrão do projeto: provavelmente um hook useSocket() ou import direto do cliente socket
  const socket = getSocketClient(); // adaptar ao padrão existente
  
  socket.on('report:pronto', (payload: { relatorioId: string; formato: string; downloadUrl: string; expiresAt: string }) => {
    setItems(prev =>
      prev.map(item =>
        item.id === payload.relatorioId
          ? { ...item, status: 'PRONTO' as const, expiresAt: payload.expiresAt }
          : item,
      ),
    );
    // Toast de notificação
    toast.success(`Relatório ${payload.formato} pronto para download!`);
  });

  return () => { socket.off('report:pronto'); };
}, []);
```

```typescript
async function handleDownload(relatorioId: string) {
  const res = await fetch(`/api/relatorios/${relatorioId}/download`);
  if (res.status === 410) { toast.error('Link expirado — solicite um novo relatório'); return; }
  if (!res.ok) { toast.error('Erro ao gerar link de download'); return; }
  const { downloadUrl } = await res.json();
  window.open(downloadUrl, '_blank', 'noopener,noreferrer');
}
```

**apps/web/src/app/(admin)/relatorios/page.tsx**

Server Component (sem `'use client'`). Next.js 15 App Router.

```typescript
import { getServerSession } from 'next-auth'; // ou auth() do Auth.js v5
import { authOptions } from '@/lib/auth';
import { ReportForm } from '@/components/relatorios/report-form';
import { ReportList } from '@/components/relatorios/report-list';

export default async function RelatoriosPage() {
  const session = await getServerSession(authOptions);
  if (!session) redirect('/login');

  // SSR da lista inicial de relatórios
  const baseUrl = process.env.API_INTERNAL_URL ?? 'http://api:4000';
  const res = await fetch(`${baseUrl}/api/relatorios?limit=20`, {
    headers: { Authorization: `Bearer ${session.accessToken}` },
    cache: 'no-store', // sempre fresco — status muda com frequência
  });
  const { items: initialItems } = res.ok ? await res.json() : { items: [] };

  return (
    <main className="p-6">
      <h1 className="font-heading text-2xl font-bold text-gray-900 mb-6">Relatórios</h1>
      
      <section className="mb-8 rounded-lg border border-gray-200 bg-white p-6 shadow-sm">
        <h2 className="text-lg font-semibold text-gray-800 mb-4">Novo Relatório</h2>
        <ReportForm onReportRequested={/* client-side handler */} />
      </section>

      <section>
        <h2 className="text-lg font-semibold text-gray-800 mb-4">Seus Relatórios</h2>
        <ReportList initialItems={initialItems} />
      </section>
    </main>
  );
}
```

Isso evita passar funções de Server para Client Component diretamente.

| window.open(downloadUrl) | URL de terceiro (Garage público) — usar noopener,noreferrer |
