    ```typescript
    'use server';
    import { redirect } from 'next/navigation';
    import { cookies } from 'next/headers';

    export async function criarEmpresaAction(_prev: unknown, formData: FormData) {
      const body = {
        nome: formData.get('nome'),
        cnpj: (formData.get('cnpj') as string).replace(/\D/g, ''),
        adminEmail: formData.get('adminEmail'),
        adminNome: formData.get('adminNome'),
        adminSenha: formData.get('adminSenha'),
      };
      const cookieStore = await cookies();
      const sessionCookie = cookieStore.getAll()
        .map(c => `${c.name}=${c.value}`).join('; ');

      const res = await fetch(`${process.env.INTERNAL_API_URL}/api/admin/empresas`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json', cookie: sessionCookie },
        body: JSON.stringify(body),
      });
      if (!res.ok) {
        const { error } = await res.json();
        return { error: error ?? 'Erro ao criar empresa' };
      }
      redirect('/admin');
    }
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → /admin/* | Qualquer usuário autenticado — middleware Next.js filtra por role |
| ImpersonateButton → cookie | Token JWE colocado em cookie de sessão do browser |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-07-07 | Elevation of Privilege | Middleware Next.js para /admin | mitigate | `auth()` verifica role antes de renderizar qualquer page do grupo (superadmin); não-SUPER_ADMIN recebe redirect para / |
| T-07-08 | Elevation of Privilege | ImpersonateButton cookie | mitigate | Cookie de impersonação tem max-age=900 (15 min) sem renovação possível; o JWE não permite refresh (sem refresh token emitido) |
| T-07-09 | Information Disclosure | fetch SSR com cookie de sessão | mitigate | Cookie repassado apenas em Server Components/Actions — nunca exposto ao cliente via props |
| T-07-10 | Tampering | criarEmpresaAction CNPJ | mitigate | CNPJ stripped de não-dígitos antes de enviar à API; validação de comprimento mínimo (14 dígitos) na API |
</threat_model>
