3. Criar `apps/api/src/middleware/auth.ts` (decisão D5 — jwtDecrypt + hkdf; Pitfall 3 salt):
```typescript
import { jwtDecrypt } from 'jose';
import { hkdf } from '@panva/hkdf';
import type { Request, Response, NextFunction } from 'express';

const COOKIE_NAME =
  process.env.NODE_ENV === 'production'
    ? '__Secure-authjs.session-token'
    : 'authjs.session-token';

async function getDerivedKey(secret: string): Promise<Uint8Array> {
  return hkdf(
    'sha256',
    secret,
    COOKIE_NAME, // salt = nome do cookie (Auth.js v5)
    `Auth.js Generated Encryption Key (${COOKIE_NAME})`,
    32,
  );
}

export async function authMiddleware(req: Request, res: Response, next: NextFunction) {
  const token =
    req.cookies?.[COOKIE_NAME] ||
    req.headers.authorization?.replace('Bearer ', '');

  if (!token) return res.status(401).json({ error: 'Não autenticado' });

  try {
    const secret = process.env.AUTH_SECRET;
    if (!secret) throw new Error('AUTH_SECRET não configurado');
    const key = await getDerivedKey(secret);
    const { payload } = await jwtDecrypt(token, key);
    req.user = {
      id: payload.sub as string,
      role: payload.role as string,
      empresaId: (payload.empresaId as string | null) ?? null,
    };
    next();
  } catch {
    return res.status(401).json({ error: 'Token inválido ou expirado' });
  }
}
```

import { prisma, createTenantClient } from '@cargo-sentinel/database';

export function tenantMiddleware(req: Request, res: Response, next: NextFunction) {

}

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| cliente/câmera → Express API | requests cruzam a fronteira; token JWE precisa ser validado antes de qualquer dado de tenant |
| req.user.empresaId → Prisma | empresaId vem do token descriptografado, nunca do body — base do isolamento de tenant |
| SUPER_ADMIN → prisma raw | bypass de tenant — só pode ocorrer quando role do token (não do cliente) for SUPER_ADMIN |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-10 | Spoofing | authMiddleware (token forjado) | mitigate | jwtDecrypt valida a integridade JWE com AUTH_SECRET; token inválido → 401 |
| T-02-11 | Elevation of Privilege | algorithm confusion (jsonwebtoken vs JWE) | mitigate | uso exclusivo de jose.jwtDecrypt (D5); jsonwebtoken proibido — proteção contra alg=none/confusion |
| T-02-12 | Elevation of Privilege | tenant escape via empresaId do body | mitigate | empresaId lido SOMENTE de req.user (token); createTenantClient injeta em 100% das queries |
| T-02-13 | Elevation of Privilege | SUPER_ADMIN bypass mal-aplicado | mitigate | ramo SUPER_ADMIN só ativa com role do token; usuário normal sem empresaId → 403 (não cai em prisma raw) |
| T-02-14 | Information Disclosure | token replay após expiração | mitigate | jwtDecrypt verifica claim exp automaticamente; token expirado → 401 |
| T-02-15 | Tampering | headers inseguros | mitigate | helmet() aplicado globalmente |
